Stack Implementation (Static Array)

#include
#include
#define SIZE 6
int stack[SIZE];
int top = -1;

int isEmpty() {
    return top == -1;
}

int isFull() {
    return top == SIZE - 1;
}

void push(int ele) {
    if (isFull()) {
        printf("stack overflow %d", ele);
        return;
    }
    stack[++top] = ele;
    printf("%d pushed to stack\n", ele);
}

int pop() {
    if (isEmpty()) {
        printf("stack underflow\n");
        exit(EXIT_FAILURE);
    }
    return stack[top--];
}
